function GetWFashReqWeapon(val){
	var txt="";
		if (val > 0){
		for(var i=0;i<FashWMCombo.length;i++){
			if (FashWMCombo[i][1]==val){
				return FashWMCombo[i][0];
				break;
			}
		}	
		var len=FashWeapMask.length;
		var max;
		var txtArr=[];
		var tai=-1;
		var x;
		for(var i=0;i<len;i++){
			//alert(Math.pow(2, (len-i-1)));
			x=FashWeapMask[len-i-1][1];
			if (x<=val){
				tai++;
				val=val-x;
				txtArr[tai]=FashWeapMask[len-i-1][0];
			}
		}
		if (tai > -1){
			txt=txtArr.join(", ");
		}
		return txt;
	}
	return txt;
}
